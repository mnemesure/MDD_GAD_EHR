def distplotvar(var_name, obj):
    var_index = list(obj.X.columns).index(var_name)
    shap = []
    for shap_val in obj.shap_vals:
        shap.append(shap_val[var_index])

    linked_vars = zip(obj.X.iloc[:,var_index], shap)
    link = sorted(linked_vars, key = lambda t: t[0])
    x_0 = []
    x_1 = []
    y_0 = []
    y_1 = []
    for i in link:
        if i[0] == 0.0:
            x_0.append(i[0])
            y_0.append(i[1])
        else:
            x_1.append(i[0])
            y_1.append(i[1])
    
    return y_0, y_1
